<ComponentDemo

      id: 'definition-tooltip',
      label: 'Definition tooltip',
    },
  ]}
  scope={{ Information }}
>
  <ComponentVariant
    id="tooltip"
    links={{

      triggerText="Tooltip label"
    >
      <button className="tooltip-trigger" type="button">
        <Information />
      </button>
    </Tooltip>

  >
